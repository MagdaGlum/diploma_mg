Задание 1

SELECT c.login,
      COUNT(o.courierId) AS orders_in_delivery
FROM Couriers AS c JOIN Orders AS o ON o.courierId = c.id 
GROUP BY c.login, o.courierId, o.inDelivery
HAVING o.inDelivery = true;


Задание 2
SELECT track,
       CASE
           WHEN finished = true THEN '2'
           WHEN cancelled = true THEN '-1'
           WHEN inDelivery = true THEN '1'
           ELSE '0'
       END
FROM Orders;
